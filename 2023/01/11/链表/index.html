#include <iostream>

class SeqList{
    int *list;
    int len;
    int maxLen;
public:
    SeqList(int n);
    ~SeqList();
    void multiInsert(int i, int n, int *item);
    void multiDel(int i, int n);
    void outPut();
};

SeqList::SeqList(int n) {
    maxLen = 1000;
    len = n;
    list = new int[maxLen];
    for(int i=0;i<n;i++)
        cin>>list[i];
}

SeqList::~SeqList() {
    delete []list;
}

void SeqList::multiInsert(int i, int n, int *item) {
    if(len+n>maxLen || i<1 || i>len+1)
        return;

    for(int j=len-1;j>=i-1;j--)
        list[j+n] = list[j];

    for(int j=i-1;j<i-1+n;j++)
        list[j] = item[j-i+1];

    len += n;
}

void SeqList::multiDel(int i, int n) {
    if(i<1 || i>len || i+n-1>len)
        return;

    for(int j=i-1+n;j<len;j++)
        list[j-n] = list[j];

    len -= n;

}

void SeqList::outPut() {
    cout<<len<<' ';
    for(int i=0;i<len;i++)
        cout<<list[i]<<' ';
    cout<<endl;
}

SeqList::SeqList(int* data, int size)
{
    maxsize = 1000;
    this->size = size;

{
    if (i <= 0 || i > size + 1) {
        cout << "error" << endl;
        return error;
    }
    auto q = &list[i - 1];
    for (auto p = &list[size - 1]; p >= q; p--)
        *(p + 1) = *p;
    *q = item;
    size++;
    list_display();
    return ok;
}

int SeqList::list_delete(int i)
{
    if (i <= 0 || i > size + 1) {
        cout << "error" << endl;
        return error;
    }
    auto q = &list[i - 1];
    for (; q <= &list[size - 1]; q++)
        *q = *(q + 1);
    list[size - 1] = 0;
    size--;
    list_display();
    return ok;
}

{
    if (i <= 0 || i > size + 1) {
        cout << "error" << endl;

int SeqList::multiinsert(int i, int n, int item[])
{
    if (i <= 0 || i > size + 1 || size + n > maxsize) {
        cout << "error" << endl;
        return error;
    }
    auto q = &list[i - 1];
    for (auto p = &list[size - 1]; p >= q; p--)
        *(p + n) = *p;
    for (int j = 0; j < n; j++) {
        *q = item[j];
        q++;
    }
    size += n;
    list_display();
    return ok;
}

int SeqList::multidelete(int i, int n)
{
    if (i <= 0 || i > size + 1 || size - n < 0) {
        cout << "error" << endl;
        return error;
    }
    auto q = &list[i - 1];
    for (; q <= &list[size - 1]; q++)
        *q = *(q + n);
    for (int j = 0; j < n; j++) {
        list[size - j - 1] = 0;
    }	
    size -= n;
    list_display();
    return ok;
}

int SeqList::list_union(SeqList& rhs)
{
    for (int i = 0; i < rhs.size; i++) {
        list[size + i] = rhs.list[i];
    }
    size += rhs.size;
    for (int i = 0; i < size; i++)
        for (int j = 1; j < size - i; j++) {
            if (list[j - 1] > list[j])
                swap(list[j - 1], list[j]);
        }
    list_display();
    return ok;
}

{
    cout << size;
    for (int i = 0; i < size; i++)
        cout << " " << list[i];

#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
struct node {
    int num, de, cai;
};
int cmp(node a, node b) {
    if ((a.de + a.cai) != (b.de + b.cai))
        return (a.de + a.cai) > (b.de + b.cai);
    else if (a.de != b.de)
        return a.de > b.de;
    else
        return a.num < b.num;
}
int main() {
    int n, low, high;
    scanf("%d %d %d", &n, &low, &high);
    vector<node> v[4];
    node temp;
    int total = n;
    for (int i = 0; i < n; i++) {
        scanf("%d %d %d", &temp.num, &temp.de, &temp.cai);
        if (temp.de < low || temp.cai < low)
            total--;
        else if (temp.de >= high && temp.cai >= high)
            v[0].push_back(temp);
        else if (temp.de >= high && temp.cai < high)
            v[1].push_back(temp);
        else if (temp.de < high && temp.cai < high && temp.de >= temp.cai)
            v[2].push_back(temp);
        else
            v[3].push_back(temp);
    }
    printf("%d\n", total);
    for (int i = 0; i < 4; i++) {
        sort(v[i].begin(), v[i].end(), cmp);
        for (int j = 0; j < v[i].size(); j++)
            printf("%d %d %d\n", v[i][j].num, v[i][j].de, v[i][j].cai);
    }
    return 0;
}

LinkList::~LinkList()
{
    ListNode *p = head;

    {
        q = p->next;
        delete p;
        p = q;
    }
    len = 0;
    head = NULL;
}

ListNode * LinkList::Index(int i)
{
    ListNode *p = head;
    while(i--)
    {
        p = p->next;

    }

    head = NULL;
}

void LinkList::CreateInTail(int n){
    len = n;
    LNode *tail=head;
    while(n--){
        int item;
        cin>>item;
        LNode *s = new LNode;
        s->data=item;
        tail->next=s;
        tail=s;
    }
}

void LinkList::outPut(){
    LNode *p=head->next;

        cout<<p->data<<" ";
        p=p->next;
    }
    cout<<endl;
}

LinkList LinkList::merge(LinkList &li){
    LinkList temp;
    temp.len = len+li.len;
    LNode *p = head->next,*q=li.head->next,*r=temp.head;
    while(p&&q){
        LNode *s=new LNode;
        if(p->data < q->data){
            s->data=p->data;
            p=p->next;
        }else{
            s->data=q->data;
            q=q->next;
        }
        r->next = s;
        r=r->next;
    }

    while(p){
        LNode *s=new LNode;
        s->data=p->data;
        r->next = s;
        r=r->next;
        p=p->next;
    }
    while(q){
        LNode *s=new LNode;
        s->data=q->data;
        r->next =s;
        r=r->next;
        q = q->next;
    }
    return temp;
}

    void add(LinkList &l){
        ListNode *p=head->next,*q=l.head->next,*pre = head;
        while(p !=NULL && q!= NULL){
            if(p->index==q->index){
                p->data+=q->data;
                ListNode *temp = q;
                if(p->data==0){
                    p=p->next;
                    delete pre->next;
                    pre->next = p;
                    q = q->next;
                }else{
                    p=p->next;
                    q=q->next;
                    pre = pre->next;
                }
                delete temp;
            }
            else if(p->index > q->index){
                pre->next = q;
                q = q->next;
                pre->next->next = p;
                pre = pre->next;
            }else{
                pre = pre->next;
                p = p->next;
            }
        }
        if(p==NULL){
            pre->next = q;
        }
        l.head->next = NULL;
    }
};

LinkList::LinkList(){
    head= new ListNode();
}

LinkList::~LinkList(){
    ListNode *p,*q;
    p=head;
    while(p !=NULL){
        q=p;
        p=p->next;
        delete q ;
    }
    head=NULL;
}

void LinkList ::LL_display(){
    ListNode *p = head->next;

    while(p){
        if(p!= head->next){
            cout<<" + ";
        }
        if(p->data == 0){
            p=p->next;
            continue;
        }else if(p->data<0){
            cout<<"("<< p->data << ")";
        }else{
            cout<<p->data;
        }
        if(p->index!=0){
            cout<<"x^";
            if(p->index<0){
                cout<<"(" << p->index<< ")";
            }else{
                cout<<p->index;
            }
        }
        p=p->next;
    }
    cout<<endl;
}
